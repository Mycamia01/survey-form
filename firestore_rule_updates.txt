rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 👤 Users can read/write their own profile
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // 📋 Surveys - allow all authenticated users
    match /surveys/{surveyId} {
      allow read, write: if request.auth != null;
    }

    // 📥 Recipients - allow all authenticated users
    match /recipients/{recipientId} {
      allow read, write: if request.auth != null;
    }

    // ✅ Responses - allow anyone to create, only authenticated can read
    match /responses/{responseId} {
      allow create: if true;
      allow read: if request.auth != null;
    }

    // 📤 Notification logs - allow all authenticated users
    match /notifications/{notificationId} {
      allow read, write: if request.auth != null;
    }

    // 🏷️ Survey categories - allow all authenticated users
    match /categories/{categoryId} {
      allow read, write: if request.auth != null;
    }
  }
}
